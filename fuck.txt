name: Auto Build

on:
    push:
        branches: [master]
    workflow_dispatch:

jobs:
    build:
        runs-on: windows-latest

        steps:
            - uses: actions/checkout@v2

            # Setup Nuget
            - name: Setup NuGet
              uses: NuGet/setup-nuget@v1.0.5

            # Setup MSBuild
            - name: Setup MSBuild
              uses: microsoft/setup-msbuild@v1.0.2

            # Setup 7-Zip
            - name: Setup 7-Zip
              run: curl https://www.7-zip.org/a/7z1900-x64.exe --output 7z.exe

            # Build the Solution
            - name: Build the Solution
              run: msbuild "Ancient Mysteries.sln" -restore /p:Configuration=Release /p:OutputPath=buildOutput

            # Pack built files into single zip
            - name: Pack built files
              run: .\7z.exe a -tzip ".\release.zip" ".\buildOutput\*"

            # Create & Update Release
            - name: Upload binaries to release
              uses: svenstaro/upload-release-action@v2
              with:
                repo_token: ${{ secrets.GITHUB_TOKEN }}
                file: D:\a\AncientMysteriesMod\AncientMysteriesMod\AncientMysteries\release.zip
                asset_name: release_${{ github.runid }}.zip
                tag: ${{ github.ref }}
                overwrite: true
                body: "Generated"
                prerelease: true
